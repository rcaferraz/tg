\subsection{Transmissão involuntária dos dados}
\begin{frame}
	\frametitle{Transmissão involuntária dos dados}
	\begin{itemize}
	 \item Método: Infecção do computador
	 \item Possibilidades:
		\begin{itemize}
		 \item Leitura de arquivos pessoais
		 \item Leitura de registro das teclas pressionadas
		 \item Leitura da lista de processos abertos
		 \item Leitura total dos dados enviados à Internet
		 \item Visualização da tela 
		 \item Visualização da imagem capturada pela \emph{webcam} 
		 \item Escuta do áudio capturado pelo microfone
		 \item Controle total sobre a máquina
		\end{itemize}
	 \item Exemplo: \emph{GhostNet}
	\end{itemize}
\end{frame}

\subsection{Interceptação dos dados em trânsito}
\begin{frame}
	\frametitle{Interceptação dos dados em trânsito}
	\begin{itemize}
	 \item Método: ``Grampear'' infraestrutura
	 \item Possibilidades:
		\begin{itemize}
		 \item Leitura total dos dados enviados à Internet
		 \item Leitura parcial dos dados enviados à Internet
			\begin{itemize}
			 \item Identificação dos destinatários
			\end{itemize}
		\end{itemize}
	 \item Exemplo: Sala da \emph{NSA} no PTT da AT\&T em 2003
	\end{itemize}
\end{frame}
